Challenge #1

Write code to convert 111011111 (base 2) into base 10


# Hint in pseudocode:
# 0. Assign n to the in base 2 to be converted to base 10
#    n = 111011111
# 1. Assume output is n in base 10, and set it to 0
#    output = 0
# 2. Convert n to a string
#    eg: nstr = str(n)
# 3. Figure out how long the string (nstr) and call it nlen
# 4. Loop through the length of the string
# 5. For each iteration of the loop:
#    (we will loop upwards, from 0 till len(nstr)-1)
#    this means that we should expect to read the digits from right to left
#    (like the Arabs)
# 5a.   Assign the character by the offset indicated by the loop variable (i.e., i)
#       Eg, x = nstr[len(nstr) - 1 - i]
        (this may take time to understand)
# 5b.   Convert x to an integer, call it y
# 5c.   Multiply and sum it up (as per the algorithm)
#       output += y * pow(2, i)
# 6.    Print output
# 7.    Rejoice!

Answer #1

Write code to convert 111011111 (base 2) into base 10


n = 111011111
output = 0
nstr = str(n)
nlen = len(nstr)

# Our ever erstwhile loop variable
i=0

while (i < nlen):

    x = nstr[nlen - 1 - i]
    y = int(x)
    output += y * pow(2, i)

# This should print 479
print output

def bin2dec(n):

    i=0
    output = 0
    nstr = str(n)
    nlen = len(nstr)

    while (i < nlen):

        x = nstr[nlen - 1 - i]
        y = int(x)
        output += y * pow(2, i)

    return output

def something2dec(n, base):

    i=0
    output = 0
    nstr = str(n)
    nlen = len(nstr)

    while (i < nlen):

        x = nstr[nlen - 1 - i]
        y = int(x)
        output += y * pow(base, i)

    return output

# Send in a base 2 number
